public void Start()
{
    while (true)
    {
        Console.Clear();
        Draw();

        var action = _nav.GetAction();

        switch (action)
        {
            case NavigationAction.Up:
                MoveUp();
                break;

            case NavigationAction.Down:
                MoveDown();
                break;

            case NavigationAction.Select:
                Select();
                break;

            case NavigationAction.Back:
                return; 
        }
    }
}
